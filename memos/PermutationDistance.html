#插入排序包
import bisect
while True:
    try:
        N=int(input().strip())
        a=[]
        for i in range(N):
            a.append(int(input().strip()))
        res=0
        lst=[a[0]]
        for i in range(1,N):
            j=bisect.bisect_left(lst,a[i])
            bisect.insort(lst,a[i])
            res+=i-j 
        print(res)
    except:
        break
